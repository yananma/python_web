
题目描述  
从上到下按层打印二叉树，同一层结点从左至右输出。每一层输出一行。  


```python 
# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    # 返回二维列表[[1,2],[4,5]]
    def Print(self, pRoot):
        if not pRoot:
            return [] 
        
        result = [] 
        queue = [pRoot, ] 
        while queue:
                res1 = [] 
                for _ in range(len(queue)):
                    temp = queue.pop(0) 
                    res1.append(temp.val) 
                    if temp.left:
                        queue.append(temp.left) 
                    if temp.right:
                        queue.append(temp.right) 
                result.append(res1) 
        return result 
```
