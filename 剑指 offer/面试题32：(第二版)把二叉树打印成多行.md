
题目描述  
从上到下按层打印二叉树，同一层结点从左至右输出。每一层输出一行。  


```python 
class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None
        

class Solution:
    def levelOrder(self, root):
        if not root:
            return []
        queue = [root, ]
        result = []
        while queue:
            res = []
            for _ in range(len(queue)):    # 遍历一层  
                temp = queue.pop(0)
                res.append(temp.val)
                if temp.left:
                    queue.append(temp.left)
                if temp.right:
                    queue.append(temp.right)
            result.append(res)
        return result


t1 = TreeNode(1)
t2 = TreeNode(2)
t3 = TreeNode(3)

t1.left = t2
t1.right = t3

s = Solution()
s.levelOrder(t1)
```
