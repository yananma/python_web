class Solution:
    def isValid(self, s):
        dicts = {")": "(", "]": "[", "}": "{"}  # 字典键值对，键是右括号，值是左括号    
        stack = []
        for char in s:
            if char not in dicts:  # 不在 dicts 的键中，一定是左括号  
                stack.append(char)  # stack 里放的都是左括号  
            else:  # 右括号
                if not stack or stack[-1] != dicts[char]:
                    return False
                stack.pop()  # 有一对儿配对成功就 pop 出来  
        return len(stack) == 0  # 最后为空，就是 True


s = Solution()
s.isValid('(){')
