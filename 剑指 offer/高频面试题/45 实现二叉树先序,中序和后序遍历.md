分别按照二叉树先序，中序和后序打印所有的节点。  

```python 
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def threeOrders(self, root):
        self.pre, self.mid, self.pos = [], [], []
        self.preOrder(root)
        self.midOrder(root)
        self.posOrder(root)
        return [self.pre, self.mid, self.pos]

    def preOrder(self, root):
        if not root:
            return None
        self.pre.append(root.val)
        self.preOrder(root.left)
        self.preOrder(root.right)

    def midOrder(self, root):
        if not root:
            return None
        self.midOrder(root.left)
        self.mid.append(root.val)
        self.midOrder(root.right)

    def posOrder(self, root):
        if not root:
            return None
        self.posOrder(root.left)
        self.posOrder(root.right)
        self.pos.append(root.val)
```
